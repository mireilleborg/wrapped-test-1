<script>

  function goTo(id){

    document.querySelectorAll('.slide').forEach(s=>s.classList.remove('active'));

    document.getElementById(id).classList.add('active');

    window.scrollTo({top:0,behavior:'smooth'});

  }


  // Reveal cards

  document.querySelectorAll('.reveal').forEach(el=>{

    el.addEventListener('click',()=>{

      el.style.filter='none';

      el.textContent = el.getAttribute('data-text');

      el.style.transition='filter 220ms';

    });

  });


  // Quiz

  document.querySelectorAll('.option').forEach(opt=>opt.addEventListener('click',()=>{

    document.querySelectorAll('.option').forEach(o=>o.classList.remove('correct'));

    if(opt.dataset.correct==='true'){

      opt.classList.add('correct');

      setTimeout(()=>goTo('s7'),600);

    } else {

      opt.style.opacity=0.6;

    }

  }));


  function previewAudio(){

    alert('Open Spotify via the button — browser audio previews vary by browser and available APIs.');

  }


  // Email-friendly link copy (or Netlify link copy)

  function shareEmail(){

    const url = location.href;

    navigator.clipboard.writeText(url).then(()=>alert('Link copied to clipboard — paste it in your message.'));

  }

  function downloadPNG(){
